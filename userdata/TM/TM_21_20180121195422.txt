
    if(!detect(FORWARD(), OBJECT.WALL)){
        if(detect(LEFT(), OBJECT.WALL)){
            move(FORWARD());
        }
        else{
            move(LEFT());
        }
    }

        }
        else{
            move(LEFT());
        }
    }
